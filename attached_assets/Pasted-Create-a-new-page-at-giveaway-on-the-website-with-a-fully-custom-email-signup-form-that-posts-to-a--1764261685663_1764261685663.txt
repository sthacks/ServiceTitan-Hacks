  e.preventDefault();

  const res = await fetch("YOUR_WEBHOOK_URL", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ email })
  });

  if (res.ok) {
    document.getElementById("form-status").innerText = "You're entered. Check your email.";
    document.getElementById("giveaway-form").reset();
  } else {
    document.getElementById("form-status").innerText = "There was an issue. Please try again.";
  }
});
</script>
